// ProducerConsumerSemaphore.java
import java.util.*;
import java.util.concurrent.Semaphore;

public class ProducerConsumerSemaphore {
    static Queue<String> buffer = new LinkedList<>();
    static final int BUFFER_SIZE = 5;

    static Semaphore empty = new Semaphore(BUFFER_SIZE);
    static Semaphore full = new Semaphore(0);
    static Semaphore mutex = new Semaphore(1);

    static class Producer extends Thread {
        int id;
        Producer(int id) { this.id = id; }
        public void run() {
            for (int i = 0; i < 5; i++) {
                try {
                    String item = "Item-" + id + "-" + i;
                    empty.acquire();
                    mutex.acquire();
                    buffer.add(item);
                    System.out.println("Producer-" + id + " produced " + item + " | Buffer: " + buffer);
                    mutex.release();
                    full.release();
                    Thread.sleep((int)(Math.random()*300));
                } catch (InterruptedException e) {}
            }
        }
    }

    static class Consumer extends Thread {
        int id;
        Consumer(int id) { this.id = id; }
        public void run() {
            for (int i = 0; i < 5; i++) {
                try {
                    full.acquire();
                    mutex.acquire();
                    String item = buffer.remove();
                    System.out.println("Consumer-" + id + " consumed " + item + " | Buffer: " + buffer);
                    mutex.release();
                    empty.release();
                    Thread.sleep((int)(Math.random()*300));
                } catch (InterruptedException e) {}
            }
        }
    }

    public static void main(String[] args) throws Exception {
        new Producer(1).start();
        new Producer(2).start();
        new Consumer(1).start();
        new Consumer(2).start();
    }
}
